import java.util.*;
import java.io.*;

/*
물고기
번호가 작은것부터 움직임
빈칸이나 다른물고기 칸으로 이동할 수 있음 (해당 칸의 물고기와 위치를 서로 바꾸는 식으로 이동)
경계를 넘거나 상어칸으로는 갈 수 없음
이동할 곳이 없으면 45도 반시계 회전하면서 갈 수 있는곳 찾음
갈 수 있는곳이 없으면 이동안함
그리고 상어한테 턴을 넘김

상어
처음에 0,0에 물고기를 먹고 그 방향을 흡수함 (그리고 물고기에 턴을 넘김)
방향에 있는 물고기중 하나의 칸으로 이동 가능
물고기를 먹고 그 물고기의 방향을 흡수함
그리고 물고기에 턴을 넘김
갈 수 있는 곳이 없으면 알고리즘 종료시킴


상어가 먹을 수 있는 물고기 번호의 합의 최댓값을 구하라 

*/
public class Main{
    public static void main(String[] args) throws IOExceptions {
        BufferedReader br = new BufferedReader(new InputStreamReader(System.in));
        BufferedWriter bw = new BufferedWriter(new OutputStreamWriter(System.out));

        int[][] fish = new int[17][3];
        int[][] map = new int[4][4]; 

        for(int i=0;i<4;i++) {
            StringTokenizer st= new StringTokenizer(br.readLine());
            for(int j=0;j<4;j++) {
                int a = Integer.parseInt(st.nextToken());
                int b = Integer.parseInt(st.nextToken()); 
                map[i][j] = a; // 물고기번호
                fish[a] = new int[]{i,j,b}; // 행,렬,크기
            }
        }

    }
}
