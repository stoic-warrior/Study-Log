```
import java.util.*;
import java.io.*;

/*
처음 상어크기 2
1초에 상하좌우 한칸 움직임

상어보다 큰 물고기칸 못지나감
같은크기 물고기칸 지나감
작은 물고기는 이동과 동시에 먹을 수 있음

더이상 먹을 수 있는 물고기 없을 때 ㅡ> 종료 (몇초동안 물고기 잡아먹으러 다녓는지 출력)
먹을 수 있는 물고기 1마리 ㅡ> 그 물고기 먹으로
2마리 이상 ㅡ> 가까운거 먹으러
가까운게 많으면 그중 위에거, 위에 많으면 그중 왼쪽거 먼저

거리 = 이동할때 지나는 칸 개수 
상어 크기와 같은 수의 물고기 먹으면 크기 1증가 (크기2면 물고기 2마리 먹고 크기 3됨)

ㅡㅡㅡㅡㅡㅡ

중간에 큰 물고기가 있으니 멘헤튼 거리로는 안되고
BFS를 해야될듯


BFS로 최소거리 반환하고
디폴트 반환하면 먹을 수 있는게 없는걸로 

 */
public class Main{

    static int[] dr = {-1,1,0,0};
    static int[] dc = {0,0,-1,1};
    static int[][] map;
    static int currSize = 2;
    static int N;
    static int sr = -1;
    static int sc = -1;
    public static void main(String[] args) throws IOException {
        BufferedReader br = new BufferedReader(new InputStreamReader(System.in));
        BufferedWriter bw = new BufferedWriter(new OutputStreamWriter(System.out));

        N = Integer.parseInt(br.readLine()); // 공간의 크기 2~20
        map = new int[N][N];
        int eatCnt = 0;
        int distSum = 0;

        for(int i=0;i<N;i++) {
            StringTokenizer st = new StringTokenizer(br.readLine());
            for(int j=0;j<N;j++) {
                map[i][j] = Integer.parseInt(st.nextToken()); // 9는 상어, 0은 빈칸, 나머지는 물고기 크기

                if(map[i][j]==9) {
                    sr = i;
                    sc = j;
                    map[i][j] = 0; // 실수노트 : 이거 안해서 처음 상어자리가 큰물고기 벽처럼 작동함 
                }
            }
        }


        // 이동거리 계속 더하고, 물고기먹은게 쌓이면 상어사이즈up 
        while(true) {
            int dist = BFS(sr,sc); // 상어위치를 매개변수로 

            if(dist==(int) 2e9) break; // 먹을 물고기 없음

            map[sr][sc] = 0;
            distSum += dist;
            eatCnt++;

            if(eatCnt==currSize) {
                eatCnt = 0;
                currSize++;
            }
        }

        bw.write(distSum+"");
        bw.flush();
        bw.close();
    }

    // 상어 출발점input, 이동거리 output
    // 상어위치 갱신 
    static int BFS(int a, int b) {
        int fishR = -1;
        int fishC = -1;
        int minDist = (int) 2e9;
        int[][] dist = new int[N][N];
        for(int i=0;i<N;i++) {
             Arrays.fill(dist[i],-1);
        }

        Queue<int[]> q = new ArrayDeque<>();

        q.offer(new int[]{a,b});
        dist[a][b] = 0;

        while(!q.isEmpty()) {
            int[] curr = q.poll();
            int r = curr[0];
            int c = curr[1];

            if(dist[r][c]>=minDist) break; // 더이상 최단거리 갱신안됨 

            for(int i=0;i<4;i++) {
                int nr = r + dr[i];
                int nc = c + dc[i]; 

                if(nr<0 || nr>=N || nc<0 || nc>=N || map[nr][nc]>currSize) continue; // 장외, 상어보다 큰 물고기

                if(dist[nr][nc]==-1) {
                    dist[nr][nc]=dist[r][c]+1;
                    q.offer(new int[]{nr,nc});

                    if(map[nr][nc]<currSize && map[nr][nc]!=0 && minDist>=dist[nr][nc]) { // 작은물고기면서 최단거리
                        
                        if(fishR==-1){ // 첫 물고기
                            minDist = dist[nr][nc];
                            fishR = nr;
                            fishC = nc;
                        }
                        else { // 최단거리 같은 다른 물고기
                            if(nr<fishR) {
                                fishR = nr;
                                fishC = nc;
                            }
                            else if (nr==fishR && nc<fishC) {
                                fishR = nr;
                                fishC = nc;
                            }
                        }
                        
                    }
                }
            }
        }

        if (fishR != -1) { // 먹을 물고기가 있는 경우에만 상어위치 갱신 
            sr = fishR;
            sc = fishC;
        }

        return minDist;
    }
}
```

- 시간복잡도 : 최악 O(N⁴)

```
import java.util.*;
import java.io.*;

/*
물고기
번호가 작은것부터 움직임
빈칸이나 다른물고기 칸으로 이동할 수 있음 (해당 칸의 물고기와 위치를 서로 바꾸는 식으로 이동)
경계를 넘거나 상어칸으로는 갈 수 없음
이동할 곳이 없으면 45도 반시계 회전하면서 갈 수 있는곳 찾음
갈 수 있는곳이 없으면 이동안함
그리고 상어한테 턴을 넘김

상어
처음에 0,0에 물고기를 먹고 그 방향을 흡수함 (그리고 물고기에 턴을 넘김)
방향에 있는 물고기중 하나의 칸으로 이동 가능
물고기를 먹고 그 물고기의 방향을 흡수함
그리고 물고기들에 턴을 넘김
갈 수 있는 곳이 없으면 알고리즘 종료시킴


상어가 먹을 수 있는 물고기 번호의 합의 최댓값을 구하라 
ㅡㅡㅡㅡ

DFS로 맵 보내면서 최댓값 갱신해야 할듯 
스택으로 하면 너무 복잡해서 재귀로 해야될듯 

*/
public class Main{

    static int[] dr = {-1,-1,0,1,1,1,0,-1}; // 0-indexed 
    static int[] dc = {0,-1,-1,-1,0,1,1,1};
    static int max = 0; // 먹은 물고기 번호 합의 최댓값 
    public static void main(String[] args) throws IOException {
        BufferedReader br = new BufferedReader(new InputStreamReader(System.in));
        BufferedWriter bw = new BufferedWriter(new OutputStreamWriter(System.out));

        int[][] fish = new int[17][3];
        int[][] map = new int[4][4]; 

        for(int i=0;i<4;i++) {
            StringTokenizer st= new StringTokenizer(br.readLine());
            for(int j=0;j<4;j++) {
                int a = Integer.parseInt(st.nextToken());
                int b = Integer.parseInt(st.nextToken()); 
                map[i][j] = a; // 물고기번호
                fish[a] = new int[]{i,j,b-1}; // 행,렬,방향 
            }
        }


        dfs(map,fish,0,0,0);

        bw.write(max+"");
        bw.flush();
        bw.close();

    }

    static void dfs(int[][] map, int[][] fish, int sr,int sc, int currSum) { // sr sc는 샤크의 위치
        // 샤크위치의 물고기를 제거하고 방향을 흡수한다. 물고기 번호는 sum해서 최대값 갱신해야 한다
        currSum += map[sr][sc];
        max = Math.max(max,currSum);
        int dir = fish[map[sr][sc]][2]; // 상어의 방향
        fish[map[sr][sc]][2] = -1;
        map[sr][sc] = 0;
        


        // 물고기 낮은번호부터 순서대로 움직인다
        for(int i=1;i<=16;i++) {
            if(fish[i][2]!=-1) { // 물고기가 살아있으면
                int r = fish[i][0];
                int c = fish[i][1];
                int d = fish[i][2];

                for(int j=0;j<8;j++) { // 8가지 방향
                    int nd = (d + j)%8;

                    int nr = r + dr[nd];
                    int nc = c + dc[nd];

                    if(nr<0 || nr>3 || nc<0 || nc>3 || (nr==sr && nc==sc)) continue;

                    if(map[nr][nc]==0) { // 빈칸
                        map[r][c] = 0;
                        map[nr][nc] = i;
                        fish[i][0] = nr;
                        fish[i][1] = nc;
                        fish[i][2] = nd;

                        break;
                    }
                    else { // 물고기끼리 위치바꾸기
                        int tmp = map[nr][nc];
                        map[nr][nc] = i;
                        fish[i][0] = nr;
                        fish[i][1] = nc;
                        fish[i][2] = nd;

                        map[r][c] = tmp;
                        fish[tmp][0] = r;
                        fish[tmp][1] = c;

                        break;
                    }
                }
            }
        }

        // 상어가 물고기칸으로 움직이고 dfs재귀, 갈곳없으면 종료
        // 방향의 1배, 2배, 3배 다 확인 
        for(int i=1;i<=3;i++) {
            int nsr = sr + i*dr[dir];
            int nsc = sc + i*dc[dir];

            if(nsr<0 || nsr>3 || nsc<0 || nsc>3) break;

            if(map[nsr][nsc]!=0) {
                dfs(copyMap(map), copyFish(fish),nsr,nsc,currSum);
            }
        }
        
        
    }

        static int[][] copyMap(int[][] map) {
        int[][] copy = new int[4][4];

        for (int i = 0; i < 4; i++) {
            copy[i] = map[i].clone();
        }

        return copy;
    }

    static int[][] copyFish(int[][] fish) {
        int[][] copy = new int[17][3];

        for (int i = 0; i <= 16; i++) {
            copy[i] = fish[i].clone();
        }

        return copy;
    }
}
```
